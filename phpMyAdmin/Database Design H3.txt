CREATE DATABASE h3;
USE h3;

CREATE TABLE accounts (
  id INT NOT NULL AUTO_INCREMENT,
  name VARCHAR (16) NOT NULL,
  PRIMARY KEY (id)
) AUTO_INCREMENT = 1000;

CREATE TABLE schools (
  id INT NOT NULL AUTO_INCREMENT,
  name VARCHAR (16) NOT NULL,
  PRIMARY KEY (id)
) AUTO_INCREMENT = 1000;

CREATE TABLE teams (
  id INT NOT NULL AUTO_INCREMENT,
  name VARCHAR (16) NOT NULL,  
  PRIMARY KEY (id)
) AUTO_INCREMENT = 1000;

CREATE TABLE events (
  id INT NOT NULL AUTO_INCREMENT,
  name VARCHAR (16) NOT NULL,  
  PRIMARY KEY (id)
) AUTO_INCREMENT = 1000;

CREATE TABLE tickets (
  id INT NOT NULL AUTO_INCREMENT,
  ticket_id INT NOT NULL,
  ticket_order INT NOT NULL,
  PRIMARY KEY (id)
) AUTO_INCREMENT = 1000;

CREATE TABLE revenue (
  id INT NOT NULL AUTO_INCREMENT,
  bo DECIMAL (9,2) NOT NULL,
  pos DECIMAL (9,2) NOT NULL,
  associations DECIMAL (9,2) NOT NULL,
  payout_date INT NOT NULL,
  PRIMARY KEY (id),
) AUTO_INCREMENT = 1000;
